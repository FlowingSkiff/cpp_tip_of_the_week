</p></details><details open><summary>Example</summary><p>

```cpp
struct foo {
  std::size_t bar;
};

int main() {
  std::cout << std::meta::offset_of(^foo::bar); // 0
  std::cout << std::meta::size_of(^foo::bar); // 8
  std::cout << std::meta::alignment_of(^foo::bar); // 8
}
```
